import csv
from datetime import datetime

# ---------- CONFIG ----------
COMBINED_FILE = "combinedMocks.csv"
EQ_FILE = "equifax-mocks-dates.csv"
TU_FILE = "transunion-mocks-dates.csv"
EXP_FILE = "experian-mocks-dates.csv"


# ---------- HELPERS ----------

def read_csv(filepath):
    with open(filepath, newline="") as f:
        return list(csv.reader(f))


def write_csv(filepath, rows):
    with open(filepath, "w", newline="") as f:
        writer = csv.writer(f)
        writer.writerows(rows)


def normalize_dates_from_combined(rows):
    dates = set()
    for r in rows:
        if r and r[0].startswith("1960-"):
            dates.add(r[0])
    return dates


def find_free_date(existing_dates):
    # Prefer December
    for day in range(1, 32):
        d = f"1960-12-{day:02d}"
        if d not in existing_dates:
            return d

    # fallback: whole year
    for month in range(1, 13):
        for day in range(1, 32):
            try:
                d = datetime(1960, month, day).strftime("%Y-%m-%d")
                if d not in existing_dates:
                    return d
            except:
                continue


def convert_formats(date_str):
    dt = datetime.strptime(date_str, "%Y-%m-%d")
    return {
        "tu": date_str,
        "eq_exp": dt.strftime("%m%d%Y")
    }


def append_or_create(rows, scenario, date_val):
    found = False

    for r in rows:
        if r and r[0].strip() == scenario:
            found = True
            dates = r[1].split("|") if len(r) > 1 else []

            if date_val not in dates:
                r[1] = r[1] + date_val + "|"
            return rows

    # Not found → create
    rows.append([scenario, f"{date_val}|"])
    return rows


def combination_exists(rows, exp, eq, tu):
    for r in rows:
        if len(r) < 4:
            continue
        if r[1] == exp and r[2] == eq and r[3] == tu:
            return True
    return False


def insert_combined(rows, new_row):
    last_index = -1

    for i, r in enumerate(rows):
        if r and r[0].startswith("1960-12-"):
            last_index = i

    insert_at = last_index + 1
    rows.insert(insert_at, new_row)

    return rows


# ---------- MAIN FUNCTION ----------

def process(EQ=None, TU=None, EXP=None):
    combined = read_csv(COMBINED_FILE)

    # Step 1: Check duplicate combination
    if combination_exists(combined, EXP, EQ, TU):
        print("Combination already exists. Doing nothing.")
        return

    # Step 2: Find free date
    existing_dates = normalize_dates_from_combined(combined)
    free_date = find_free_date(existing_dates)

    print("Using date:", free_date)

    formats = convert_formats(free_date)

    # Step 3: Update bureau files

    if EQ:
        eq_rows = read_csv(EQ_FILE)
        eq_rows = append_or_create(eq_rows, EQ, formats["eq_exp"])
        write_csv(EQ_FILE, eq_rows)

    if TU:
        tu_rows = read_csv(TU_FILE)
        tu_rows = append_or_create(tu_rows, TU, formats["tu"])
        write_csv(TU_FILE, tu_rows)

    if EXP:
        exp_rows = read_csv(EXP_FILE)
        exp_rows = append_or_create(exp_rows, EXP, formats["eq_exp"])
        write_csv(EXP_FILE, exp_rows)

    # Step 4: Update combined
    new_row = [
        free_date,
        EXP or "",
        EQ or "",
        TU or ""
    ]

    combined = insert_combined(combined, new_row)
    write_csv(COMBINED_FILE, combined)

    print("✅ Successfully updated all files.")


# ---------- USAGE ----------

process(
    EQ="EQSuccess_ValidFico",
    TU="TU_Bureau_EWS_Hit",
    EXP="ExpSuccess_ValidFico"
)
