document.querySelectorAll("input[name='inputType']").forEach(radio => {
    radio.addEventListener("change", () => {
        const type = document.querySelector("input[name='inputType']:checked").value;
        document.getElementById("jsonInput").style.display =
            type === "json" ? "block" : "none";
        document.getElementById("fileInput").style.display =
            type === "file" ? "block" : "none";
    });
});

document.getElementById("runBtn").addEventListener("click", async () => {
    const type = document.querySelector("input[name='inputType']:checked").value;
    const formData = new FormData();

    if (type === "json") {
        const jsonText = document.getElementById("jsonInput").value;
        formData.append("json_data", jsonText);
    } else {
        const fileInput = document.getElementById("fileInput");
        if (fileInput.files.length === 0) {
            alert("Select a file");
            return;
        }
        formData.append("file", fileInput.files[0]);
    }

    try {
        const response = await fetch("http://127.0.0.1:8000/run-tests", {
            method: "POST",
            body: formData
        });

        const data = await response.json();
        document.getElementById("status").innerText =
            "Status: " + JSON.stringify(data);
    } catch (err) {
        document.getElementById("status").innerText = "Error: " + err;
    }
});

from fastapi import FastAPI, File, UploadFile, Form
from typing import Optional
from playwright.sync_api import sync_playwright
import pandas as pd
import json
import tempfile

app = FastAPI()

@app.post("/run-tests")
async def run_tests(
    json_data: Optional[str] = Form(None),
    file: Optional[UploadFile] = File(None)
):
    data = None

    # Case 1: JSON input
    if json_data:
        data = json.loads(json_data)

    # Case 2: Spreadsheet input
    elif file:
        with tempfile.NamedTemporaryFile(delete=False) as tmp:
            tmp.write(await file.read())
            tmp_path = tmp.name

        df = pd.read_excel(tmp_path)
        data = df.to_dict(orient="records")

    else:
        return {"error": "Provide either JSON or file"}

    # Example: use first record
    first_record = data[0]

    with sync_playwright() as p:
        browser = p.chromium.launch(headless=True)
        context = browser.new_context()
        page = context.new_page()

        page.goto("https://example.com/form")
        page.fill("input[name='name']", str(first_record.get("name", "")))
        page.click("button[type='submit']")

        browser.close()

    return {
        "status": "done",
        "records_processed": len(data)
    }
