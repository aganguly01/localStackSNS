popup.js

document.querySelectorAll("input[name='inputType']").forEach(radio => {
    radio.addEventListener("change", () => {
        const type = document.querySelector("input[name='inputType']:checked").value;
        document.getElementById("jsonInput").style.display =
            type === "json" ? "block" : "none";
        document.getElementById("fileInput").style.display =
            type === "file" ? "block" : "none";
    });
});

document.getElementById("runBtn").addEventListener("click", async () => {
    const type = document.querySelector("input[name='inputType']:checked").value;
    const formData = new FormData();

    if (type === "json") {
        const jsonText = document.getElementById("jsonInput").value;
        formData.append("json_data", jsonText);
    } else {
        const fileInput = document.getElementById("fileInput");
        if (fileInput.files.length === 0) {
            alert("Select a file");
            return;
        }
        formData.append("file", fileInput.files[0]);
    }

    try {
        const response = await fetch("http://127.0.0.1:8000/run-tests", {
            method: "POST",
            body: formData
        });

        const data = await response.json();
        document.getElementById("status").innerText =
            "Status: " + JSON.stringify(data);
    } catch (err) {
        document.getElementById("status").innerText = "Error: " + err;
    }
});
