CREATE OR REPLACE TABLE mytime_stg AS
SELECT *
FROM (
    SELECT *,
           ROW_NUMBER() OVER (
               PARTITION BY invoice_id
               ORDER BY updated_at DESC
           ) AS rn
    FROM mytime_raw
)
WHERE rn = 1;

Select * from mytime_stg;

CREATE OR REPLACE TABLE mytime_mart AS
SELECT
    location_id,
    business_date,
    COUNT(DISTINCT invoice_id) AS total_invoices,
    SUM(amount) AS total_amount,
    SUM(amount) / COUNT(DISTINCT invoice_id) AS avg_invoice_amount
FROM mytime_stg
GROUP BY location_id, business_date;

Select * from mytime_mart;
